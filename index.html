'updateOccurrence' : 'addOccurrence'; const payload = { ...formData, action: action }; if (currentEditId) { payload.id = currentEditId; } try { const response = await fetch(SCRIPT_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); if (!response.ok) { const errorText = await response.text(); throw new Error(`HTTP error! Status: ${response.status}, Mensagem: ${errorText.substring(0, 200)}`); } const data = await response.json(); if (data && data.status === "success") { return data; } else { const errorMsg = data?.message || "Erro ao processar ocorrência"; throw new Error(errorMsg); } } catch (error) { console.error('Erro em sendFormData:', error); throw error; } } // Função para carregar e exibir os registros na tabela - VERSÃO CORRIGIDA async function loadRegistrosTable() { registrosTableBody.innerHTML = `Carregando registros...`; try { const url = `${SCRIPT_URL}?action=getAllOccurrences`; const data = await safeFetch(url); if (data && data.status === "success" && Array.isArray(data.occurrences)) { if (data.occurrences.length === 0) { registrosTableBody.innerHTML = `Nenhum registro encontrado.`; return; } registrosTableBody.innerHTML = ""; // Limpar antes de adicionar data.occurrences.forEach(occurrence => { const row = document.createElement("tr"); row.innerHTML = ` ${occurrence.date} ${occurrence.professor} ${occurrence.class} ${occurrence.student} ${occurrence.irregularities} ${occurrence.description || `Nenhuma observação` } `; registrosTableBody.appendChild(row); }); } else { const errorMsg = data?.message || "Erro ao carregar registros"; registrosTableBody.innerHTML = `Erro ao carregar registros: ${errorMsg}`; showNotification("error", `Erro ao carregar registros: ${errorMsg}`); } } catch (error) { console.error("Erro ao carregar registros:", error); registrosTableBody.innerHTML = `Erro ao carregar registros: ${error.message || "Erro desconhecido"}`; showNotification("error", `Erro ao carregar registros: ${error.message || "Erro desconhecido"}`); } }
